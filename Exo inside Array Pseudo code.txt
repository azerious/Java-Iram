	Déclarations :

		maximum <- get_max(Array)

Fin de main()

Fonction get_max(entier Array[]) : entier

	Déclarations :
		
		entier i <- 0
		entier max <- Array[0]

	Instructions : 
	
		TANT QUE (i < Array.length)
			SI (Array[i] > max) ALORS
				max <- Array[i];
			Fin SI
			i++;
		Fin TANT QUE
		
		RETOURNER max;

Fin get_max()
